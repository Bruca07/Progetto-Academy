drop table if exists studenti cascade;
create table studenti (
    id integer generated by default as identity,
    data_nascita date,
    cognome varchar(255) not null,
    corso_laurea varchar(255),
    email varchar(255),
    nome varchar(255) not null,
    primary key (id)
);
